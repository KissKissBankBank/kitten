## Import

```js
import { LinkBox } from '@kisskissbankbank/kitten/src/components/box/link-box'
```

## Usage

#### Default

```js
<LinkBox />
```

### With icon

```js
<LinkBox>
  <LinkBox.Icon>
    <Icon />
  </LinkBox.Icon>
</LinkBox>
```

### With text

```js
<LinkBox>
  <LinkBox.Text>
    <Text>foobar</Text>
  </LinkBox.Text>
</LinkBox>
```

### Link props

```js
<LinkBox
  href="#anchor"
  isExternal
  linkProps={{ aria-hidden: true }}
/>
```
