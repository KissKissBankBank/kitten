# LinkBox

<Canvas>
  <Story name="Default" story={stories.Default} />
</Canvas>

## Import

```js
import { LinkBox } from '@kisskissbankbank/kitten'
```

## Usage

#### Default

```js
<LinkBox />
```

### With icon

```js
<LinkBox>
  <LinkBox.Icon>
    <Icon />
  </LinkBox.Icon>
</LinkBox>
```

### With text

```js
<LinkBox>
  <LinkBox.Text>
    <Text>foobar</Text>
  </LinkBox.Text>
</LinkBox>
```

### Link props

```js
<LinkBox
  href="#anchor"
  isExternal
  linkProps={{ aria-hidden: true }}
/>
```

### Variant props

<Canvas>
  <Story name="WithVariant" story={stories.WithVariant} />
</Canvas>


```js
<LinkBox variant="andromeda">
  <LinkBox.Text>
    <Text>foobar</Text>
  </LinkBox.Text>

```

### BackgroundColor props

```js
<LinkBox backgroundColor="#fff">
  <LinkBox.Text>
    <Text>foobar</Text>
  </LinkBox.Text>
</LinkBox>
```

### TextColor props

```js
<LinkBox textColor="#222">
  <LinkBox.Text>
    <Text>foobar</Text>
  </LinkBox.Text>
```
